#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "请输入队员数量：";
    cin >> n;

    int rounds = n - 1;  // 循环赛轮次数
    int matches_per_round = n / 2;  // 每轮比赛对数

    // 生成所有可能的比赛对决组合，并按规则安排场次顺序
    for (int round = 1; round <= rounds; round++) {
        cout << "第" << round << "轮比赛：" << endl;
        for (int i = 1; i <= matches_per_round; i++) {
            int player1 = (round - 1 + i) % (n - 1);
            int player2 = (n - 1 - i + round) % (n - 1);
            if (i == 1) {
                player2 = n - 1;
            }
            cout << "队员" << player1 + 1 << " vs 队员" << player2 + 1 << endl;
        }
    }

    return 0;
}
